### 结构体

- 定义

```go
type person struct {
	Name string
	Age int
}
func main() {
	p := person{
		Name: "lppx",
        Age:90,
	}
	p.Age = 10
	fmt.Println(p) // {lppx 10}
}
通常用法 在赋值的时候 加上取地址符号&,p := &person{}
type person struct {
	Name string
	Age int
}
func main() {
	p := &person{
		Name: "lppx",
		Age:90,
	}
	p.Age = 10
	fmt.Println(p) // {lppx 10}
	A(p)
	fmt.Println(p) // &{knight 10}
}
func A(per *person)  {
	per.Name = "knight"
}
```

- 匿名结构、嵌套结构

```go
type person struct {
	Name string
	Age int
	Contact struct{
		Phone, City string
	}
}
func main() {
	p := &person{
		Name: "lppx",
		Age:90,
	}
	p.Contact.Phone = "199"
	p.Contact.City = "GX"
	fmt.Println(p) // &{lppx 90 {199 GX}}
}
```

- 结构比较 

  类型相同，字段及值相同才相同

```go
type person struct {
	Name string
	Age int
}
func main() {
	a := person{Name:"lpx", Age: 19}
	b := person{Name:"lpx", Age: 20}
	fmt.Println(a==b) // false
}
```

- 类似继承

```go
type human struct {
	Sex int
}
type teacher struct {
	human
	Name string
	Age int
}
type student struct {
	human
	Name string
	Age int
}
func main() {
	a := teacher{Name:"lpx", Age: 19, human:human{Sex: 0}}
	b := student{Name:"lpx", Age: 20, human:human{Sex: 1}}
	fmt.Println(a, b) // {{0} lpx 19} {{1} lpx 20}
	a.human.Sex = 100
	fmt.Println(a) // {{100} lpx 19}
	b.Sex = 888
	fmt.Println(b) // {{888} lpx 20}
}
```

- 练习

```go
type A struct {
	B
	Name string
}
type B struct {
	Name string
}
func main() {
	a := A{Name:"A", B:B{Name:"B"}}
	fmt.Println(a.Name, a.B.Name) // A B
}
```
